#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MAXN=300001;
ll a[MAXN],sum[MAXN],invn[MAXN],invN[MAXN];
ll n;
const ll MOD=998244353;
ll quickpow(ll a,ll b)
{
    ll e=1;
    while(b)
    {
        if(b&1)e=(a*e)%MOD;
        a=(a*a)%MOD;
        b>>=1;
    }
    return e;
}
ll quickinv(ll a)
{
    return quickpow(a,MOD-2);
}
void init()
{
    sum[0]=0;
    invn[0]=invN[0]=1;
    ll _n=quickinv(n);
    for(int i=1;i<=3;i++)
    {
        invn[i]=(invn[i-1]*_n)%MOD;
    }
    ll _N=(n+1)*_n%MOD;
    for(int i=1;i<=n;i++)
    {
        invN[i]=(invN[i-1]*_N)%MOD;
        cin>>a[i];
        sum[i]=(sum[i-1]+a[i])%MOD;
    }
}
ll f(ll i)
{
    ll ans1=0,ans2=0;
    ans1=(((i*a[i])%MOD*invn[2])%MOD*invN[i-1])%MOD;
    ans2=(((i*sum[i-1])%MOD*invn[3])%MOD*invN[i-2])%MOD;
    return (ans1+ans2)%MOD;
}
int main()
{
    cin>>n;
    init();
    ll ans=0;
    for(int i=1;i<=n;i++)
    {
        ans=(ans+f(i))%MOD;
    }
    cout<<ans<<endl;
    system("pause");
    return 0;

}
